  20:    function cursor_update_objStoreRecords(objName, newRecords, txnMode, txnTimeout, txnCommit) {
  21:        var db = null;
  22:        try
  23:        {
  24:        var indexName = "ckids";
  25:        var dbreq = window.indexedDB.open(indexedDBName);
  26:        dbreq.onsuccess = function (event) {
  27:            db = event.result;
  28:            var txn = db.transaction(null, txnMode, txnTimeout);
  29:            if (txn) 
  30:            {
  31:                txn.oncomplete = function () {
  32:                    output_trace("transaction completed.");
  33:                    db.close();
  34:                }
  35:                txn.onabort = function () {
  36:                    output_trace("transaction aborted.");
  37:                    db.close();
  38:                }
  39:                txn.ontimeout = function () {
  40:                    output_trace("transaction timeout.");
  41:                    db.close();
  42:                }
  43:  
  44:                var store = txn.objectStore(objName);
  45:                if (store) 
  46:                {
  47:                    var index = store.Index(indexName);
  48:                    var cursorReq = store.openCursor();
  49:                    cursorReq.onsuccess = function (ev1) {
  50:                        var cur = ev1.result;
  51:                        cursor_update_record(cur, txn, newRecords, 0, txnCommit);
  52:                    }
  53:                    cursorReq.onerror = function (ev1) {
  54:                        output_trace("Failed to open cursor. Error: " + ev.message);
  55:                    }
  56:                }
  57:            }
  58:        }
  59:        dbreq.onerror = function (event) {
  60:            output_trace("indexedDB.open Error: " + event.message);
  61:        }
  62:      }
  63:      catch (e)
  64:      {
  65:          output_trace(e.message);
  66:          if (db)
  67:              db.close();
  68:      }
  69:    }

  71:    function cursor_update_record(cur, txn, newRecords, i, commitT) 
  72:    {
  73:        try 
  74:        {
  75:            if (i < newRecords.length)
  76:            {
  77:                var rec = newRecords[i];
  78:                i++;
  79:                var curMoveReq = cur.move();
  80:                curMoveReq.onsuccess = function (evt) {
  81:                    if (cur.key) {
  82:                        try {
  83:                            var updateReq = cur.update(rec);
  84:                            updateReq.onsuccess = function (ev) {
  85:                                PrintRecord("KIDSTORE", rec);
  86:                                cursor_update_record(cur, txn, newRecords, i, commitT);
  87:                            }
  88:                            updateReq.onerror = function (ev) {
  89:                                output_trace("failed to update record at cursor.");
  90:                            }
  91:                        }
  92:                        catch (e1) {
  93:                            output_trace(e1.message);
  94:                        }
  95:                    }
  96:                    else { //no more record
  97:                        if (commitT == true)
  98:                            txn.db.close(); //txn.commit();
  99:                        else
 100:                            txn.abort();
 101:                    }
 102:                }
 103:                curMoveReq.onerror = function (evt) {
 104:                    output_trace("Index key cursor move Error or no more record: " + evt.message);
 105:                }
 106:            }
 107:            else if (i == newRecords.length) 
 108:            {
 109:              if(commitT==true)
 110:                  txn.db.close() ; // txn.commit();
 111:              else
 112:                  txn.abort();
 113:            }
 114:        }
 115:      catch (e) 
 116:        {
 117:            output_trace(e.message);
 118:            txn.abort();
 119:        }
 120:    }
