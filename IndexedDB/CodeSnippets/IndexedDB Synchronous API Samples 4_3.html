   2: function index_openCursor() 
   3: {
   4:     var objStoreName = "KIDSTORE";
   5:     var indexName = "cnumber";
   6:  
   7:     try {
   8:         var db = window.indexedDBSync.open(indexedDBName);
   9:         if (db) {
  10:             var objStore = db.openObjectStore(objStoreName, 1); //read_only
  11:             if (objStore) {
  12:                 var openedIndex = objStore.index(indexName);
  13:                 if (openedIndex) 
  14:                 {
  15:                     var curCursor = openedIndex.openCursor();
  16:                     if (curCursor) 
  17:                     {
  18:                         while (curCursor.move()) 
  19:                         {
  20:                             output_trace("index.openCursor():  (" + 
                                   curCursor.key + "," + curCursor.value + ")");
  21:                         }
  22:                     }
  23:                     else {
  24:                         output_trace("Failed to open cursor.");
  25:                     }
  26:                 }
  27:                 else {
  28:                     output_trace("Failed to open Index " + indexName + " .");
  29:                 }
  30:             }
  31:             db.close();
  32:         }
  33:     }
  34:     catch (e) {
  35:         output_trace("Error: " + e.message);
  36:     }
  37: }
